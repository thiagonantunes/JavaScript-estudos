git clone // entrar na pasta, abrir terminal git clone + URL
git revert <id commit>
git reset -- hard <id commit anterior>
git commit --amend -m "<mensagem>" // alterar mensagem do commit
git branch <nome> // ramificações do projeto    -d <nome branch> deletar    -m <renomear> 
git checkout <nome branch> // modificar branch   -b cria novo branch
git switch -c <nome branch> // criar novo branch e alterar para ele 
git merge <nome branch> a partir da main// unir dois branch
